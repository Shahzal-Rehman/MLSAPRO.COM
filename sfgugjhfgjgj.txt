        function login() {
            var clientId = "9ec16646-a913-4fdb-8249-08fb1d98b99d"; // Your client ID
            var redirectUri = encodeURIComponent("https://team.mlsapro.com"); // Redirect URL after login

            var scope = "user.read"; // Adjust scope as needed
            var url = `${authority}?client_id=${clientId}&response_type=code&redirect_uri=${redirectUri}&scope=${scope}`;
            window.location.href = url;
        }

        function signOut() {
            var clientId = "9ec16646-a913-4fdb-8249-08fb1d98b99d"; // Your client ID

            var url = authority + "?post_logout_redirect_uri=" + postLogoutRedirectUri;
            window.location.href = url;
        }
